1. Napiši program koji ispisuje najveći broj između dva unesena broja.
 
#include <stdio.h>

int main() {
    int a, b;
    printf("Unesite prvi broj: ");
    scanf("%d", &a);
    printf("Unesite drugi broj: ");
    scanf("%d", &b);

    int max = a > b ? a : b;
    printf("Najveci broj je: %d\n", max);

    return 0;
}
